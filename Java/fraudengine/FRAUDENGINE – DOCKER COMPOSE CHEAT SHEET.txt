FRAUDENGINE – DOCKER COMPOSE CHEAT SHEET

--START APP
docker compose up -d
docker-compose up -d --build
docker-compose --env-file docker/.env up -d --build

--STOP APP (SAFE, KEEP DATA)
docker compose stop

--STOP AND REMOVE CONTAINERS
docker compose down

--FULL RESET (DELETE DATABASE DATA)
docker compose down -v

--RESTART EVERYTHING
docker compose restart

--RESTART ONLY APP
docker compose restart fraudengine

--VIEW ALL LOGS
docker compose logs -f

--VIEW APP LOGS
docker compose logs -f fraudengine

--VIEW POSTGRES LOGS
docker compose logs -f postgres

--BUILD JAR
./mvnw clean package

--REBUILD IMAGE + START
docker compose up -d --build

--CHECK RUNNING CONTAINERS
docker ps
